Install following dependencies before running example. Suggest to create a new conda environment and install dependencies
	> numpy, scipy : conda install numpy scipy matplotlib
	> sympy : conda install sympy
	> networkx : pip install networkx==2.3
  	> tqdm : conda install -c conda-forge tqdm
  	> ipykernel : conda install ipykernel
  	> ffmpeg : conda install -c conda-forge ffmpeg
	> ortools:  python -m pip install --upgrade --user ortools
